        // Global variables
        let filteredData = [...shippingData];
        let currentFilters = {
            itemType: '',
            size: '',
            region: '',
            search: '',
            priceSort: ''
        };

        // DOM elements
        const itemTypeSelect = document.getElementById('itemType');
        const itemSizeSelect = document.getElementById('itemSize');
        const regionSelect = document.getElementById('region');
        const priceSortSelect = document.getElementById('priceSort');
        const searchBox = document.getElementById('searchBox');
        const clearFiltersBtn = document.getElementById('clearFilters');
        const resultsContainer = document.getElementById('resultsContainer');
        const resultCount = document.getElementById('resultCount');
        const noResults = document.getElementById('noResults');

        // Initialize the application
        function init() {
            updateSizeOptions();
            renderResults();
            setupEventListeners();
        }

        // Event listeners
        function setupEventListeners() {
            itemTypeSelect.addEventListener('change', handleItemTypeChange);
            itemSizeSelect.addEventListener('change', handleFilterChange);
            regionSelect.addEventListener('change', handleFilterChange);
            priceSortSelect.addEventListener('change', handleFilterChange);
            searchBox.addEventListener('input', handleSearchChange);
            clearFiltersBtn.addEventListener('click', clearAllFilters);
        }

        // Handle item type change
        function handleItemTypeChange() {
            currentFilters.itemType = itemTypeSelect.value;
            updateSizeOptions();
            applyFilters();
        }

        // Handle other filter changes
        function handleFilterChange(event) {
            const filterName = event.target.id;
            let value = event.target.value;
            
            if (filterName === 'itemType') {
                currentFilters.itemType = value;
            } else if (filterName === 'itemSize') {
                currentFilters.size = value;
            } else if (filterName === 'region') {
                currentFilters.region = value;
            } else if (filterName === 'priceSort') {
                currentFilters.priceSort = value;
            }
            
            applyFilters();
        }

        // Handle search change
        function handleSearchChange() {
            currentFilters.search = searchBox.value.toLowerCase();
            applyFilters();
        }

        // Update size options based on selected item type
        function updateSizeOptions() {
            const selectedItemType = itemTypeSelect.value;
            itemSizeSelect.innerHTML = '<option value="">အားလုံး</option>';
            
            if (selectedItemType) {
                const sizes = [...new Set(shippingData
                    .filter(item => item.itemType === selectedItemType)
                    .map(item => item.size))];
                
                sizes.forEach(size => {
                    const option = document.createElement('option');
                    option.value = size;
                    option.textContent = size;
                    itemSizeSelect.appendChild(option);
                });
            } else {
                // Show all sizes if no item type selected
                const allSizes = [...new Set(shippingData.map(item => item.size))];
                allSizes.forEach(size => {
                    const option = document.createElement('option');
                    option.value = size;
                    option.textContent = size;
                    itemSizeSelect.appendChild(option);
                });
            }
        }

        // Apply all filters
        function applyFilters() {
            filteredData = shippingData.filter(item => {
                // Item type filter
                if (currentFilters.itemType && item.itemType !== currentFilters.itemType) {
                    return false;
                }
                
                // Size filter
                if (currentFilters.size && item.size !== currentFilters.size) {
                    return false;
                }
                
                // Region filter
                if (currentFilters.region && item.region !== currentFilters.region) {
                    return false;
                }
                
                // Search filter
                if (currentFilters.search) {
                    const searchTerm = currentFilters.search;
                    return (
                        item.itemType.toLowerCase().includes(searchTerm) ||
                        item.size.toLowerCase().includes(searchTerm) ||
                        item.region.toLowerCase().includes(searchTerm) ||
                        item.location.toLowerCase().includes(searchTerm) ||
                        item.price.toString().includes(searchTerm)
                    );
                }
                
                return true;
            });
            
            // Apply sorting
            if (currentFilters.priceSort) {
                filteredData.sort((a, b) => {
                    if (currentFilters.priceSort === 'asc') {
                        return a.price - b.price;
                    } else {
                        return b.price - a.price;
                    }
                });
            }
            
            renderResults();
        }

        // Render results
        function renderResults() {
            resultsContainer.innerHTML = '';
            
            if (filteredData.length === 0) {
                noResults.classList.remove('hidden');
                resultCount.textContent = '0 ရလဒ်';
                return;
            }
            
            noResults.classList.add('hidden');
            resultCount.textContent = `${filteredData.length} ရလဒ်`;
            
            filteredData.forEach((item, index) => {
                const card = createResultCard(item, index);
                resultsContainer.appendChild(card);
            });
        }

        // Create result card
        function createResultCard(item, index) {
            const card = document.createElement('div');
            card.className = 'price-card bg-white border border-gray-200 rounded-lg p-4 shadow-sm fade-in';
            card.style.animationDelay = `${index * 0.1}s`;
            
            const itemIcon = getItemIcon(item.itemType);
            
            card.innerHTML = `
                <div class="flex items-center justify-between mb-3">
                    <div class="flex items-center">
                        <i class="${itemIcon} text-2xl text-blue-600 mr-3"></i>
                        <div>
                            <h3 class="font-semibold text-gray-800">${item.itemType}</h3>
                            <p class="text-sm text-gray-600">${item.size}</p>
                        </div>
                    </div>
                    <div class="text-right">
                        <p class="text-2xl font-bold text-green-600">${item.price.toLocaleString()}</p>
                        <p class="text-sm text-gray-500">ကျပ်</p>
                    </div>
                </div>
                
                <div class="border-t pt-3 space-y-2">
                    <div class="flex items-center justify-between text-sm">
                        <span class="text-gray-600">ဒေသ:</span>
                        <span class="font-medium">${item.region}</span>
                    </div>
                    <div class="flex items-center justify-between text-sm">
                        <span class="text-gray-600">တည်နေရာ:</span>
                        <span class="font-medium">${item.location}</span>
                    </div>
                    ${item.dimensions ? `
                    <div class="flex items-center justify-between text-sm">
                        <span class="text-gray-600">အရွယ်အစား:</span>
                        <span class="font-medium">${item.dimensions}</span>
                    </div>
                    ` : ''}
                    ${item.weight ? `
                    <div class="flex items-center justify-between text-sm">
                        <span class="text-gray-600">အလေးချိန်:</span>
                        <span class="font-medium">${item.weight}</span>
                    </div>
                    ` : ''}
                </div>
            `;
            
            return card;
        }

        // Get icon for item type
        function getItemIcon(itemType) {
            const iconMap = {
                'မြင်းကျားအိတ်': 'fas fa-suitcase',
                'ကော်ပုံး Square': 'fas fa-box',
                'ကော်ပုံး Circle': 'fas fa-circle',
                'ခရီးဆောင်အိတ်': 'fas fa-luggage-cart',
                'ဂျပ်ဖါ': 'fas fa-shopping-bag',
                'ပီနံ(ဖြူ)': 'fas fa-cube',
                'ရေခဲသတ္တာ': 'fas fa-snowflake',
                'TV': 'fas fa-tv'
            };
            return iconMap[itemType] || 'fas fa-box';
        }

        // Clear all filters
        function clearAllFilters() {
            currentFilters = {
                itemType: '',
                size: '',
                region: '',
                search: '',
                priceSort: ''
            };
            
            itemTypeSelect.value = '';
            itemSizeSelect.value = '';
            regionSelect.value = '';
            priceSortSelect.value = '';
            searchBox.value = '';
            
            updateSizeOptions();
            applyFilters();
        }

        // Initialize the application when DOM is loaded
        document.addEventListener('DOMContentLoaded', init);
    </script>
